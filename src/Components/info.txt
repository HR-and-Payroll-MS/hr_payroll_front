components are reusable UI building blocks
these are my small , reusable pieces --buttons, forms, cards, modals, headers, etc
they don't depend on a specific page - they can be used anywhere in your app

************** example 1 button *****************

export default function Button({children, onClick,type = "button"}){
    return (
        <button
            onClick={onClick}
            type={type}
            className="............"
        >
        {children}
        </button>
    )
}
